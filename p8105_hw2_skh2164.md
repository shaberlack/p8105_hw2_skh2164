pols_month = separate(pols_month, mon, c("year", "month", "day"), "-")|>
  mutate(month = case_match(
    month,
    "01" ~ "January",
    "02" ~ "February",
    "03" ~ "March",
    "04" ~ "April",
    "05" ~ "May",
    "06" ~ "June",
    "07" ~ "July",

    "09" ~ "September",
    "10" ~ "October",
    "11" ~ "November",
    "12" ~ "December")) |>
